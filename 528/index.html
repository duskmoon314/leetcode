class Solution {
public:
    vector<int> sum;

    Solution(vector<int>& w): sum(move(w)) {
        partial_sum(sum.begin(), sum.end(), sum.begin());
    }

    int pickIndex() {
        int r = rand() % sum.back() + 1;
        return lower_bound(sum.begin(), sum.end(), r) - sum.begin();
    }
};
