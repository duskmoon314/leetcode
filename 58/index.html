class Solution {
public:
    int lengthOfLastWord(string s) {
        int len = 0;
        for (auto c = s.rbegin(); c != s.rend(); ++c){
            if (*c == ' ') {
                if (len == 0) {
                    continue;
                }
                break;
            }
            len += 1;
        }
        return len;
    }
};
